-- 1. ENUM TYPES
-- =====================================================
do $$
begin
  if not exists (select 1 from pg_type where typname = 'app_user_status_enum') then
    create type app_user_status_enum as enum ('active', 'inactive');
  end if;

  if not exists (select 1 from pg_type where typname = 'app_user_role_enum') then
    create type app_user_role_enum as enum ('customer', 'admin');
  end if;

  if not exists (select 1 from pg_type where typname = 'app_gender_enum') then
    create type app_gender_enum as enum ('male', 'female', 'other');
  end if;

  if not exists (select 1 from pg_type where typname = 'product_gender_enum') then
    create type product_gender_enum as enum ('men', 'women', 'unisex');
  end if;

  if not exists (select 1 from pg_type where typname = 'product_type_enum') then
    create type product_type_enum as enum (
      'shirt', 'tshirt', 'blouse', 'pants', 'jeans', 'jacket',
      'hoodie', 'dress', 'skirt', 'bag', 'shoes', 'accessory'
    );
  end if;

  if not exists (select 1 from pg_type where typname = 'fit_type_enum') then
    create type fit_type_enum as enum ('slim', 'regular', 'relaxed', 'oversized');
  end if;

  if not exists (select 1 from pg_type where typname = 'product_status_enum') then
    create type product_status_enum as enum ('active', 'inactive');
  end if;

  if not exists (select 1 from pg_type where typname = 'size_enum') then
    create type size_enum as enum ('xs', 's', 'm', 'l', 'xl', 'xxl', 'free_size');
  end if;

  if not exists (select 1 from pg_type where typname = 'order_status_enum') then
    create type order_status_enum as enum (
      'pending', 'confirmed', 'processing', 'shipped', 'delivered', 'cancelled'
    );
  end if;

  if not exists (select 1 from pg_type where typname = 'delivery_status_enum') then
    create type delivery_status_enum as enum ('pending', 'in_transit', 'delivered');
  end if;

  if not exists (select 1 from pg_type where typname = 'payment_status_enum') then
    create type payment_status_enum as enum ('pending', 'paid', 'failed');
  end if;

  if not exists (select 1 from pg_type where typname = 'payment_method_enum') then
    create type payment_method_enum as enum ('card', 'fpx', 'ewallet', 'cash_on_delivery');
  end if;
end $$;

create table if not exists public.styles (
  style_id uuid primary key default gen_random_uuid(),
  style_name text not null unique,
  created_at timestamptz not null default now()
);

create table if not exists public.categories (
  category_id uuid primary key default gen_random_uuid(),
  category_name text not null unique,
  created_at timestamptz not null default now()
);

create table if not exists public.users (
  user_id uuid primary key references auth.users(id) on delete cascade,
  user_name text not null default 'User',
  user_email text not null unique,
  user_gender app_gender_enum,
  user_phone text,
  user_profile_pic text,
  height_cm numeric(5,2) check (height_cm is null or height_cm between 100 and 250),
  weight_kg numeric(5,2) check (weight_kg is null or weight_kg between 20 and 300),
  preferred_style_id uuid references public.styles(style_id) on delete set null,
  status app_user_status_enum not null default 'active',
  role app_user_role_enum not null default 'customer',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.products (
  product_id uuid primary key default gen_random_uuid(),
  category_id uuid not null references public.categories(category_id) on delete restrict,
  product_name text not null unique,
  product_description text,
  product_gender product_gender_enum not null default 'unisex',
  product_type product_type_enum not null,
  fit_type fit_type_enum default 'regular',
  color text,
  material text,
  product_price numeric(10,2) not null check (product_price >= 0),
  product_status product_status_enum not null default 'active',
  product_pic1 text,
  product_pic2 text,
  product_pic3 text,
  product_pic4 text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.product_styles (
  product_id uuid not null references public.products(product_id) on delete cascade,
  style_id uuid not null references public.styles(style_id) on delete cascade,
  primary key (product_id, style_id)
);

create table if not exists public.quantities (
  quantity_id bigserial primary key,
  product_id uuid not null references public.products(product_id) on delete cascade,
  size size_enum not null,
  product_stock integer not null default 0 check (product_stock >= 0),
  product_sold integer not null default 0 check (product_sold >= 0),
  unique (product_id, size)
);

create table if not exists public.wishlists (
  wishlist_id uuid primary key default gen_random_uuid(),
  user_id uuid not null references public.users(user_id) on delete cascade,
  product_id uuid not null references public.products(product_id) on delete cascade,
  added_time timestamptz not null default now(),
  unique (user_id, product_id)
);

create table if not exists public.cart (
  cart_id uuid primary key default gen_random_uuid(),
  user_id uuid not null references public.users(user_id) on delete cascade,
  product_id uuid not null references public.products(product_id) on delete cascade,
  quantity_id bigint not null references public.quantities(quantity_id) on delete restrict,
  quantity integer not null check (quantity > 0),
  added_time timestamptz not null default now(),
  unique (user_id, quantity_id)
);

create table if not exists public.addresses (
  address_id uuid primary key default gen_random_uuid(),
  user_id uuid not null references public.users(user_id) on delete cascade,
  address_name text,
  recipient_name text not null,
  phone text not null,
  address_line1 text not null,
  address_line2 text,
  city text not null,
  state text not null,
  post_code text not null,
  country text not null default 'Malaysia',
  is_default boolean not null default false,
  created_at timestamptz not null default now()
);

create unique index if not exists uq_addresses_one_default
on public.addresses(user_id)
where is_default = true;

create table if not exists public.deliveries (
  delivery_id uuid primary key default gen_random_uuid(),
  address_id uuid not null references public.addresses(address_id) on delete cascade,
  delivery_fee numeric(10,2) not null default 0 check (delivery_fee >= 0),
  delivery_status delivery_status_enum not null default 'pending',
  estimated_date date,
  delivered_date date,
  created_at timestamptz not null default now()

create table if not exists public.orders (
  order_id uuid primary key default gen_random_uuid(),
  user_id uuid not null references public.users(user_id) on delete cascade,
  delivery_id uuid references public.deliveries(delivery_id) on delete set null,
  order_date timestamptz not null default now(),
  orders_status order_status_enum not null default 'pending',
  order_subtotal numeric(10,2) not null default 0 check (order_subtotal >= 0),
  created_at timestamptz not null default now()
);

create table if not exists public.order_details (
  order_detail_id uuid primary key default gen_random_uuid(),
  order_id uuid not null references public.orders(order_id) on delete cascade,
  product_id uuid not null references public.products(product_id) on delete restrict,
  quantity_id bigint not null references public.quantities(quantity_id) on delete restrict,
  quantity integer not null check (quantity > 0),
  unit_price numeric(10,2) not null check (unit_price >= 0)
);

create table if not exists public.payment_methods (
  method_id uuid primary key default gen_random_uuid(),

  method_type payment_method_enum not null,

  payment_id uuid primary key default gen_random_uuid(),
  order_id uuid not null references public.orders(order_id) on delete cascade,
  user_id uuid not null references public.users(user_id) on delete cascade,
  tax numeric(10,2) not null default 0 check (tax >= 0),
  discount numeric(10,2) not null default 0 check (discount >= 0),
  total_amount numeric(10,2) not null check (total_amount >= 0),
  payment_method payment_method_enum not null,
  payment_status payment_status_enum not null default 'pending',
  payment_date timestamptz,
  created_at timestamptz not null default now()

  recommendation_id bigserial primary key,

-- Create function to automatically create user profile when auth user signs up
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
BEGIN
  INSERT INTO public.users (
    user_id,
    user_email,
    user_name,
    status,
    role,
    created_at,
    updated_at
  ) VALUES (
    new.id,
    new.email,
    COALESCE(new.raw_user_meta_data->>'full_name', 'User'),
    'active',
    'customer',
    now(),
    now()
  )
  ON CONFLICT (user_id) DO NOTHING;
  
  RETURN new;
END;
$$;

-- Create trigger for automatic user profile creation
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW
  EXECUTE FUNCTION public.handle_new_user();

create or replace view public.v_product_catalogue as
select
  p.product_id,
  c.category_name,
  p.product_name,
  p.product_description,
  p.product_gender,
  p.product_type,
  p.fit_type,
  p.color,
  p.material,
  p.product_price,
  p.product_status,
  p.product_pic1,
  p.product_pic2,
  p.product_pic3,
  p.product_pic4,
  coalesce(
    array_agg(distinct s.style_name) filter (where s.style_name is not null),
    '{}'::text[]
  ) as style_tags,
  coalesce(
    array_agg(distinct q.size::text) filter (where q.product_stock > 0),
    '{}'::text[]
  ) as available_sizes
from public.products p
join public.categories c
  on c.category_id = p.category_id
left join public.product_styles ps
  on ps.product_id = p.product_id
left join public.styles s
  on s.style_id = ps.style_id
left join public.quantities q
  on q.product_id = p.product_id
group by
  p.product_id,
  c.category_name,
  p.product_name,
  p.product_description,
  p.product_gender,
  p.product_type,
  p.fit_type,
  p.color,
  p.material,
  p.product_price,
  p.product_status,
  p.product_pic1,
  p.product_pic2,
  p.product_pic3,
  p.product_pic4;

grant select on public.v_product_catalogue to anon, authenticated;

create or replace function public.recommend_outfit_set(
  p_height_cm numeric default null,
  p_weight_kg numeric default null,
  p_style_name text default null,
  p_gender product_gender_enum default null
)
returns table (
  category_name text,
  product_id uuid,
  product_name text,
  product_price numeric,
  recommended_size text,
  matched_styles text[],
  score numeric
)
language plpgsql
security definer
set search_path = public
as $$
declare
  v_height numeric;
  v_weight numeric;
  v_gender product_gender_enum;
  v_style_id uuid;
  v_bmi numeric;
  v_sizes size_enum[];
begin
  select
    coalesce(p_height_cm, u.height_cm),
    coalesce(p_weight_kg, u.weight_kg),
    coalesce(
      p_gender,
      case
        when u.user_gender = 'male' then 'men'::product_gender_enum
        when u.user_gender = 'female' then 'women'::product_gender_enum
        else 'unisex'::product_gender_enum
      end
    )
  into v_height, v_weight, v_gender
  from public.users u
  where u.user_id = auth.uid();

  if v_height is null or v_weight is null then
    raise exception 'Height and weight are required for recommendation.';
  end if;

  select s.style_id
  into v_style_id
  from public.styles s
  where lower(s.style_name) = lower(
    coalesce(
      p_style_name,
      (
        select s2.style_name
        from public.users u2
        left join public.styles s2
          on s2.style_id = u2.preferred_style_id
        where u2.user_id = auth.uid()
      )
    )
  )
  limit 1;

  v_bmi := v_weight / power(v_height / 100.0, 2);

  if v_bmi < 18.5 then
    v_sizes := array['xs', 's']::size_enum[];
  elsif v_bmi < 23 then
    v_sizes := array['s', 'm']::size_enum[];
  elsif v_bmi < 27.5 then
    v_sizes := array['m', 'l']::size_enum[];
  else
    v_sizes := array['l', 'xl', 'xxl']::size_enum[];
  end if;

  return query
  with scored as (
    select
      c.category_name,
      p.product_id,
      p.product_name,
      p.product_price,
      coalesce(
        (
          select q1.size::text
          from public.quantities q1
          where q1.product_id = p.product_id
            and q1.product_stock > 0
            and (q1.size = any(v_sizes) or q1.size = 'free_size')
          order by case q1.size
            when 'xs' then 1
            when 's' then 2
            when 'm' then 3
            when 'l' then 4
            when 'xl' then 5
            when 'xxl' then 6
            when 'free_size' then 7
          end
          limit 1
        ),
        'out_of_stock'
      ) as recommended_size,
      coalesce(
        array_agg(distinct s.style_name) filter (where s.style_name is not null),
        '{}'::text[]
      ) as matched_styles,
      (
        (
          case
            when v_style_id is not null and exists (
              select 1
              from public.product_styles ps2
              where ps2.product_id = p.product_id
                and ps2.style_id = v_style_id
            ) then 0.50
            else 0
          end
        )
        +
        (
          case
            when exists (
              select 1
              from public.quantities q2
              where q2.product_id = p.product_id
                and q2.product_stock > 0
                and (q2.size = any(v_sizes) or q2.size = 'free_size')
            ) then 0.30
            else 0
          end
        )
        +
        coalesce(
          (
            select 0.20 * (
              sum(q3.product_sold)::numeric /
              nullif(sum(q3.product_sold + q3.product_stock), 0)
            )
            from public.quantities q3
            where q3.product_id = p.product_id
          ),
          0
        )
      )::numeric(6,4) as score
    from public.products p
    join public.categories c
      on c.category_id = p.category_id
    left join public.product_styles ps
      on ps.product_id = p.product_id
    left join public.styles s
      on s.style_id = ps.style_id
    where p.product_status = 'active'
      and (v_gender is null or p.product_gender in (v_gender, 'unisex'))
      and (
        v_style_id is null
        or exists (
          select 1
          from public.product_styles psx
          where psx.product_id = p.product_id
            and psx.style_id = v_style_id
        )
      )
    group by
      c.category_name,
      p.product_id,
      p.product_name,
      p.product_price
  ),
  ranked as (
    select
      *,
      row_number() over (
        partition by category_name
        order by score desc, product_price asc, product_name asc
      ) as rn
    from scored
    where score > 0
  )
  select
    category_name,
    product_id,
    product_name,
    product_price,
    recommended_size,
    matched_styles,
    score
  from ranked
  where rn = 1
    and category_name in ('Top', 'Bottom', 'Outerwear', 'Dress', 'Accessories')
  order by case category_name
    when 'Top' then 1
    when 'Bottom' then 2
    when 'Outerwear' then 3
    when 'Dress' then 4
    when 'Accessories' then 5
    else 99
  end;
end;
$$;

grant execute on function public.recommend_outfit_set(numeric, numeric, text, product_gender_enum)
to authenticated;

create or replace function public.save_ai_recommendation_log(
  p_height_cm numeric,
  p_weight_kg numeric,
  p_style_name text,
  p_product_ids uuid[]
)
returns bigint
language plpgsql
security definer
set search_path = public
as $$
declare
  v_style_id uuid;
  v_rec_id bigint;
begin
  select style_id
  into v_style_id
  from public.styles
  where lower(style_name) = lower(p_style_name)
  limit 1;

  insert into public.ai_recommendations (
    user_id,
    height_cm,
    weight_kg,
    requested_style_id,
    recommended_products
  )
  values (
    auth.uid(),
    p_height_cm,
    p_weight_kg,
    v_style_id,
    to_jsonb(coalesce(p_product_ids, array[]::uuid[]))
  )
  returning recommendation_id into v_rec_id;

  return v_rec_id;
end;
$$;

    when p.product_name ilike '%Ghost%' then 'Streetwear'
    when p.product_name ilike '%Floral%' then 'Chic'
    when p.product_name ilike '%Ribbon%' then 'Chic'
    when p.product_name ilike '%Puff%' then 'Chic'
    when p.product_name ilike '%Vintage%' then 'Vintage'
    when p.product_name ilike '%Sneakers%' then 'Sporty'
    when p.product_name ilike '%Cap%' then 'Sporty'
    else 'Casual'
  end
on conflict do nothing;

insert into public.product_styles (product_id, style_id)
select
  p.product_id,
  s.style_id
from public.products p
join public.styles s
  on s.style_name = case
    when p.product_type in ('dress','blouse','skirt') then 'Japanese'
    when p.product_type in ('shirt','tshirt') then 'Minimalist'
    when p.product_type in ('pants','jeans','hoodie','jacket') then 'Casual'
    when p.product_type in ('bag','shoes','accessory') then 'European'
    else 'Casual'
  end
on conflict do nothing;
